        <div class="room-gen">
            <input type="text" id="roomId" value="" readonly>
            <button class="btn-generate" onclick="generateRoom()">🎲 生成</button>
            <button class="btn-call" onclick="joinRoom()" id="joinBtn" disabled>入室</button>
        </div>
        
        <div id="mainApp" style="display:none">
            <div id="status">準備中...</div>
            
            <div class="controls">
                <button class="btn-call" onclick="makeCall()">📞 電話する</button>
                <button class="btn-control" onclick="toggleMute()">🔊 ミュート</button>
                <button class="btn-control" onclick="toggleVideo()">📹 カメラ</button>
                <button class="btn-end" onclick="hangup()">📴 切断</button>
            </div>
            
            <div class="videos" id="videos">
                <video id="localVideo" autoplay muted></video>
                <video id="remoteVideo" autoplay></video>
            </div>
            
            <div class="chat-area">
                <div id="messages"></div>
                <input type="text" id="chatInput" placeholder="チャット..." onkeypress="handleEnter(event)">
                <button onclick="sendChat()">送信</button>

    <script src="https://unpkg.com/peerjs@1.4.7/dist/peerjs.min.js"></script>
    <script>
        let peer, localStream, currentCall, dataConnection;
        let roomId = '';
        let isMuted = false, videoOn = true;

        peer = new Peer('', {
            host: 'peerjs-broker-2.herokuapp.com',
            port: 443,
            secure: true
        });

        function generateRoom() {
            roomId = Math.floor(Math.random() * 9000) + 1000;
            document.getElementById('roomId').value = roomId;
            document.getElementById('joinBtn').disabled = false;
            addMessage('🎲 新しい部屋生成: ' + roomId);
        }

        function joinRoom() {
            roomId = document.getElementById('roomId').value;
            if(roomId.length !== 4) return alert('4桁確認！');
            
            document.getElementById('mainApp').style.display = 'block';
            document.getElementById('status').textContent = 'カメラ/マイク許可中...';
            
            navigator.mediaDevices.getUserMedia({video: true, audio: true})
                .then(stream => {
                    localStream = stream;
                    document.getElementById('localVideo').srcObject = stream;
                    document.getElementById('videos').style.display = 'grid';
                    document.getElementById('status').textContent = `✅ ${roomId}準備完了！`;
                })
                .catch(() => alert('📷🔊許可して！'));

            // 着信
            peer.on('call', call => {
                currentCall = call;
                call.answer(localStream);
                call.on('stream', stream => {
                    document.getElementById('remoteVideo').srcObject = stream;
                    document.getElementById('status').textContent = '📞 通話中';
                });
            });

            // データ接続
            dataConnection = peer.connect(roomId);
            dataConnection.on('open', () => console.log('チャット接続OK'));
            dataConnection.on('data', data => {
                if(data.type === 'chat') addMessage('相手: ' + data.text);
            });
        }

        function makeCall() {
            if(!localStream) return alert('入室して！');
            currentCall = peer.call(roomId, localStream);
            document.getElementById('status').textContent = '📞 発信中...';
            
            currentCall.on('stream', stream => {
                document.getElementById('remoteVideo').srcObject = stream;
                document.getElementById('status').textContent = '📞 通話中';
            });
        }

        function hangup() {
            if(currentCall) currentCall.close();
            if(localStream) localStream.getTracks().forEach(t => t.stop());
            if(dataConnection) dataConnection.close();
            location.reload();
        }

        function toggleMute() {
            isMuted = !isMuted;
            localStream.getAudioTracks()[0].enabled = !isMuted;
        }

        function toggleVideo() {
            videoOn = !videoOn;
            localStream.getVideoTracks()[0].enabled = videoOn;
        }

        function sendChat() {
            let msg = document.getElementById('chatInput').value.trim();
            if(!msg || !dataConnection) return;
            dataConnection.send({type: 'chat', text: msg});
            addMessage('私: ' + msg);
            document.getElementById('chatInput').value = '';
        }

        function handleEnter(e) {
            if(e.key === 'Enter') sendChat();
        }

        function addMessage(text) {
            let div = document.createElement('div');
            div.textContent = `${new Date().toLocaleTimeString()}: ${text}`;
            div.className = 'msg';
            document.getElementById('messages').appendChild(div);
            div.scrollIntoView();
        }

        // 初回生成
        generateRoom();
    </script>
